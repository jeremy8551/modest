package ${packageName};

/**
 * 工程配置
 */
public class ${className} {

    /**
     * 返回构件的 groupId
     *
     * @return groupId
     */
    public static String getGroupID() {
        return "${groupId}";
    }

    /**
     * 返回构件的 artifactId
     *
     * @return artifactId
     */
    public static String getArtifactID() {
        return "${artifactId}";
    }

    /**
     * 返回构件的 version
     *
     * @return version
     */
    public static String getVersion() {
        return "${version}";
    }

}
